#include <stdlib.h>

// header[5] seems to be some width (3,6 for longer strips)
// Pixel data seems to start at data0[19]
// dataN[0] seems to be N (ex data0[0] = 0, data[1] = 1, etc.)
//
// No font or any other information on header or commands
//
// Maximum buffer size : 500 + buffer id

// -------------------------------------------------------------------------------
// Header
// -------------------------------------------------------------------------------

// Seems to contain fixed magic sequence, two 'hash' numbers
// and a 'strip lendth' field

// 501, 127
uint8_t header_p[9] = {
    0xff, 0xf0, 0x12, 0x34, // magic
    0x70, 0x02, 0x00, 0x00, // nr of pixel bytes +24
    0xa7 // 167
};

// 501, 167
uint8_t header_pp[9] = {
    0xff, 0xf0, 0x12, 0x34,
    0x98, 0x02, 0x00, 0x00, // nr of pixel bytes +24
    0xcf // 207
};

// 501, 407
uint8_t header_ppp[9] = {
    0xff, 0xf0, 0x12, 0x34,
    0x88, 0x03, 0x00, 0x00, // nr of pixel bytes +24
    0xc0 // 192
};

// 501, 407
uint8_t header_pppppp_small[9] = {
    0xff, 0xf0, 0x12, 0x34,
    0x88, 0x03, 0x00, 0x00, // nr of pixel bytes +24
    0xc0 // 192
};

// 501, 501, 501, 127
uint8_t header_pppppp[9] = {
    0xff, 0xf0, 0x12, 0x34,
    0x58, 0x06, 0x00, 0x00, // nr of pixel bytes +24
    0x93 // 147
};

// -------------------------------------------------------------------------------
// Buffer head commands
// -------------------------------------------------------------------------------

// Second command is length :
// (2*9mm margin included, length in mm = L*0.07952)
// Equation : Y = 0.07952*X + 18.26

uint8_t head_p[] = {
    0x1b, 0x73 /* maybe margin ?? */, 0x9a, 0x02, 0x00, 0x00,
    0x1b, 0x44, 0x01, 0x02,
          0x96, 0x00, 0x00, 0x00, // 150 (30mm) = nr of pixel bytes/4 ==> number of lines
          0x20, 0x00, 0x00, 0x00
};

uint8_t head_pp[] = {
    0x1b, 0x73, 0x9a, 0x02, 0x00, 0x00,
    0x1b, 0x44, 0x01, 0x02,
          0xa0, 0x00, 0x00, 0x00, // 160 (31mm)
          0x20, 0x00, 0x00, 0x00
};

uint8_t head_ppp[] = {
    0x1b, 0x73, 0x9a, 0x02, 0x00, 0x00,
    0x1b, 0x44, 0x01, 0x02,
          0xdc, 0x00, 0x00, 0x00, // 220 (36mm)
          0x20, 0x00, 0x00, 0x00

uint8_t head_pppppp_small[] = {
    0x1b, 0x73, 0x9a, 0x02, 0x00, 0x00,
